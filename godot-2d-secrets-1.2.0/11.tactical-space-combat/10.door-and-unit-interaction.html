Door and unit Interaction

In the previous lesson, we coded unit movement, but we left out unit-door interactions.

Our doors should open and close when units pass through them. We’ll add this missing feature in this lesson.

We’ll also have the units pause briefly while the door opens.

Updating the door scene

We already created the Door.tscn scene a few lesons ago. We’ll update it to include a timer that pauses for a short while the units that pass by, before opening.

Our door is an area that’s part of the door node group.

Open the scene and add a Timer node. We will use that to add a delay to the units as they wait for doors to open.

Set the Timer’s One Shot property to true so it doesn’t loop and lower its Wait Time to 0.3.

The most important part to remember is that our Sprite node has two frames: one for the closed door and another for the open door.

We don’t have any advanced animations. We will just set the appropriate frame when the time comes for units to move through the doors.

The code

Let’s start with the door code. Assign a script to the Door node and save it at res://TacticalSpaceCombat/Ship/Rooms/Door.gd.

We start with properties. The door should have an open and closed state, so we define a boolean to keep track of that.

We also want to track units touching the door’s area as the door shouldn’t close while characters walk through it.

extends Area2D

## Emitted when the door opened.
signal opened

## If `true`, the door is open, and units can pass through it. We'll define the
## setter function in a moment to control this behavior.
var is_open := false setget set_is_open

## Keeps track of the number of units passing this door right now.
var _units := 0

onready var sprite: Sprite = $Sprite
onready var timer: Timer = $Timer

We’ll use the opened signal to know when to resume unit movement after the door Timer reaches zero.

As for _units, we keep track of the units passing the door at any given moment so we know when to set is_open to false. Let’s continue:

func _ready() -> void:
    # I like to write as few functions as possible when connecting signals.
    # Here, I connect two signals to the same function and bind a boolean value.
    # This allows me to group more logic in a single place.
    connect("area_entered", self, "_on_area_entered_exited", [true])
    connect("area_exited", self, "_on_area_entered_exited", [false])
    # And here, I directly connect the signal to the `is_open` property's
    # setter function. There again, this limits indirection in GDScript code.
    timer.connect("timeout", self, "set_is_open", [true])

We use the trick of connecting two signals to the same function again. We pass the extra argument true or false to distinguish between the signals.

Since we use a setter function, set_is_open(), we can directly connect the Timer signal to it to update the state. When the Timer reaches zero, it’ll call set_is_open(true).

func _on_area_entered_exited(area: Area2D, has_entered: bool) -> void:
    if area.is_in_group("unit"):
        # Add or subtract 1 to `_units` based on `has_entered`.
        _units += 1 if has_entered else -1

        # We need to meet two conditions to open or close the door. With
        # `match`, we check them both at once.
        match [_units, has_entered]:
            # If there's no unit passing through the door and the last one just
            # exited the area, we close the door.
            [0, false]:
                self.is_open = false

            # If the first unit is entering the door, it's closed, so we start
            # the timer before opening it.
            [1, true]:
                timer.start()

We close the door only when _units is 0 and has_entered is false, meaning when all units have left the door Area2D space.

Otherwise, we start the timer only if the number of units passing the door is exactly 1 and has_entered is true, meaning there’s only one unit that entered the door area.

Thus, if more than one unit passes at the same time, the door will remain open.

And finally, we need to add our setter function:

## Updates `is_open` and the door's sprite. Emits the `opened` signal
## when `is_open` is `true`.
func set_is_open(value: bool) -> void:
    is_open = value
    if is_open:
        sprite.frame = 1
        emit_signal("opened")
    else:
        sprite.frame = 0

tool
extends Node2D

var _slots := {}

onready var tilemap: TileMap = $TileMap
onready var rooms: Node2D = $Rooms
onready var doors: Node2D = $Doors
onready var units: Node2D = $Units


func _ready() -> void:
    call("_ready_editor_hint" if Engine.editor_hint else "_ready_not_editor_hint")


func _ready_editor_hint() -> void:
    for room in rooms.get_children():
        room.setup(tilemap)


func _ready_not_editor_hint() -> void:
    for unit in units.get_children():
        for door in doors.get_children():
            door.connect("opened", unit, "set_is_walking", [true])

        var position_map := tilemap.world_to_map(unit.path_follow.position)
        _slots[position_map] = unit

    for room in rooms.get_children():
        room.setup(tilemap)
        for point in room:
            tilemap.set_cellv(point, 0)
    tilemap.setup(rooms, doors)


func _unhandled_input(event: InputEvent) -> void:
    if not event.is_action_pressed("right_click"):
        return

    for unit in get_tree().get_nodes_in_group("selected-unit"):
        var point1: Vector2 = tilemap.world_to_map(unit.path_follow.position)
        for room in get_tree().get_nodes_in_group("selected-room"):
            var point2: Vector2 = room.get_slot(_slots, unit)
            if is_inf(point2.x):
                break

            var path: Curve2D = tilemap.find_path(point1, point2)
            Utils.erase_value(_slots, unit)
            _slots[point2] = unit
            unit.walk(path)

class_name Unit
extends Path2D


const COLORS := {"default": Color("323e4f"), "selected": Color("3ca370")}

export var speed := 150

var is_walking: bool setget set_is_walking

onready var path_follow: PathFollow2D = $PathFollow2D
onready var area_unit: Area2D = $PathFollow2D/AreaUnit


func _ready() -> void:
    area_unit.connect("area_entered", self, "_on_AreaUnit_area_entered")
    area_unit.modulate = COLORS.default
    self.is_walking = false


func _on_AreaUnit_area_entered(area: Area2D) -> void:
    if area.is_in_group("door") and not area.is_open:
        self.is_walking = false


func _process(delta: float) -> void:
    path_follow.offset += speed * delta
    if path_follow.offset >= curve.get_baked_length():
        self.is_walking = false

func walk(path: Curve2D) -> void:
    if path.get_point_count() == 0:
        return
    curve = path
    curve.add_point(path_follow.position, Vector2.ZERO, Vector2.ZERO, 0)
    path_follow.offset = 0
    self.is_walking = true


func set_is_walking(value: bool) -> void:
    is_walking = value
    set_process(is_walking)
